```bash
# Install Oxlint, ESLint, and compatibility plugins
npm install --save-dev oxlint eslint eslint-plugin-oxlint eslint-plugin-storybook

# Or with yarn
yarn add --dev oxlint eslint eslint-plugin-oxlint eslint-plugin-storybook

# Or with pnpm
pnpm add --save-dev oxlint eslint eslint-plugin-oxlint eslint-plugin-storybook
```
